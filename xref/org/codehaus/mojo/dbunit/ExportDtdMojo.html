27  import org.apache.maven.plugin.MojoExecutionException;
28  import org.apache.maven.plugin.MojoFailureException;
29  import org.dbunit.ant.Export;
30  import org.dbunit.ant.Query;
31  import org.dbunit.ant.Table;
32  import org.dbunit.database.DatabaseSequenceFilter;
33  import org.dbunit.database.IDatabaseConnection;
34  import org.dbunit.dataset.FilteredDataSet;
35  import org.dbunit.dataset.IDataSet;
36  import org.dbunit.dataset.xml.FlatDtdDataSet;
37  
38  import java.io.File;
39  import java.io.FileOutputStream;
40  import java.io.OutputStreamWriter;
41  import java.io.Writer;
42  
43  /**
44   * Execute DbUnit Export DTD operation
45   *
46   * @author <a href="mailto:gdyangyu@gmail.com">Yu Yang</a>
47   * @version $Id: ExportDtdMojo.java 9171 2014-11-14 18:21:11Z Yu Yang $
48   * @goal exportDtd
49   */
50  public class ExportDtdMojo
51          extends AbstractDbUnitMojo {
52      /**
53       * Location of exported DTD file
54       *
55       * @parameter expression="${dest}" default-value="${basedir}/src/test/resources/dataset.dtd"
56       */
57      protected File dest;
58  
59      /**
60       * Encoding of exported data.
61       *
62       * @parameter expression="${encoding}" default-value="${project.build.sourceEncoding}"
63       */
64      protected String encoding;
65  
66  
67      public void execute()
68              throws MojoExecutionException, MojoFailureException {
69          if (skip) {
70              this.getLog().info("Skip export execution");
71              return;
72          }
73  
74          super.execute();
75  
76          try {
77              //dbunit require dest directory is ready
78              dest.getParentFile().mkdirs();
79  
80              IDatabaseConnection connection = createConnection();
81              try {
82                  IDataSet dataSet = new FilteredDataSet(new DatabaseSequenceFilter(connection),
83                          connection.createDataSet());
84                  Writer out = new OutputStreamWriter(new FileOutputStream(dest), encoding);
85                  FlatDtdDataSet.write(dataSet, out);
86  
87              } finally {
88                  connection.close();
89              }
90          } catch (Exception e) {
91              throw new MojoExecutionException("Error executing export", e);
92          }
93  
94      }
95  }
